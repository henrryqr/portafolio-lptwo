def sumar_numeros(a, b):
    return a + b
 
def main():
    num1 = int(input("Primer número: "))
    num2 = int(input("Segundo número: "))
    resultado = sumar_numeros(num1, num2)
    print(f"La suma es: {resultado}")
 
if __name__ == "__main__":
    main()
